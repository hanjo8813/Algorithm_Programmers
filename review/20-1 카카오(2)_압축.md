```python
def solution(msg):
    dic_list = [chr(i+65) for i in range(26)]
    index_list = []
    i =0

    while i!=len(msg):
        w = msg[i]
        while 1:
            if w in dic_list:
                pre = dic_list.index(w)+1
            else:
                index_list.append(pre)
                dic_list.append(w)
                break
            i += 1
            if i == len(msg):
                index_list.append(pre)
                break
            w += msg[i]

    return index_list
```
