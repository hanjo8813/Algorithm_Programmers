```python
from collections import deque

def solution(priorities, location):
    p_list = deque([[i,p] for i, p in enumerate(priorities)])
    priorities = deque(sorted(priorities, reverse=True))
    cnt = 0
    while 1:
        temp = p_list.popleft()
        if priorities[0] != temp[1]:
            p_list.append(temp)
        else:
            cnt+=1
            priorities.popleft()
            if temp[0] == location:
                break
    return cnt
```
